DATABASE TABLES INFORMATION
NAME , PRIMARY & FORIEGN KEY 


1. ADMIN 
   ID AUTO_INCREMENT (PRIMARY KEY) , 
   USERNAME , 
   PASSWORD
  
2. MAID 
   ID AUTO_INCREMENT (PRIMARY KEY) , NAME , USERNAME , PASSWORD , GENDER , DOB , 
   MOBILE_NO , EMAIL_ID , ADDRESS , ADHAR_CARD , POLICE_VERIFICATION_CERTIFICATE , 
   EXPERIENCE , RATING , IMAGES  

3. USER 
   ID AUTO_INCREMENT (PRIMARY KEY) , NAME , USERNAME , PASSWORD, 
   GENDER , DOB , MOBILE_NO , EMAIL_ID, ADDRESS , ADHAR_CARD , IMAGES, 
   

4. SERVICES 
   ID AUTO_INCREMENT (PRIMARY KEY) , SERVICE , CHARGES

5. BOOKING_INFO
   ID AUTO_INCREMENT (PRIMARY KEY) , 
   MAID_ID REFERENCE (MAID INFO TABLE) ID (FORIEGN KEY), 
   USER_ID REFERENCE (USER INFO TABLE) ID (FORIEGN KEY), 
   SERVICE_ID REFERENCE (SERVICES TABLE) ID (FORIGN KEY) ,
   START_DAY INT , 
   END_DAY INT ,
   CHARGES_PERDAY
   TOTAL_CHARGES INT

6. MAID_REVIEW 
   ID AUTO_INCREMENT (PRIMARY KEY) , 
   MAID_ID REFERENCE (MAID INFO TABLE) ID (FORIEGN KEY),  
   USER_ID REFERENCE (USER INFO TABLE) ID (FORIEGN KEY) ,
   RATING  , 
   DESCRIPTION

7. CONTACT_US
   ID AUTO_INCREMENT (PRIMARY KEY) , 
   NAME , 
   MOBILE_NO , 
   EMAIL_ID , 
   CITY , 
   MESSAGE
