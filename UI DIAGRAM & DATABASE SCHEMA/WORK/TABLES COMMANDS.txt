CREATE DATABASE BookMyMaid

1. create table Admin (
		ID int primary key auto_increment , 
		UserName  varchar (30) not null , 
		PassWord varchar (30) not null
);

2. create table Maid ( 
		ID int primary key auto_increment , 
		Name  varchar (30) not null  ,
		UserName varchar (30) not null, 
		PassWord varchar (30) not null , 
		Gender Varchar (10) not null , 
		DOB varchar (30) not null ,
		Mobile_No varchar (30) not null , 
		Email_Id varchar(30) not null, 
		Address  varchar (100) not null , 
		Adhar_Card varchar (30) not null , 
		Police_Verification_Certificate  varchar (30) not null , 
		Experience int , 
		Rating varchar(30)  , 
		Images varchar (70) 
);

3. create table User ( 
		ID int primary key auto_increment , 
		Name  varchar (30) not null  ,
		UserName varchar (30) not null,	
		PassWord varchar (30) not null , 
		Gender Varchar (10) not null , 
		Mobile_No varchar (30) not null , 
		Email_Id varchar(30) not null, 
		Address  varchar (100) not null ,
		Adhar_Card varchar (30) not null , 
		Images varchar (70) 
);

4. create table Services ( 
		ID int Primary key auto_increment , 
		Service varchar (30) not null , 
		Charges_PerDay int 
);

5. create table booking_info (
 		ID int primary key auto_increment , 
		Maid_Id int, 
		User_ID int ,
		Service_ID int , 
		Time varchar(30) , 
		Date varchar(30) 
);

6. create table Maid_Review (
		ID int primary key auto_increment , 
		Maid_ID int , 
		User_ID int , 
		Rating varchar(30) ,
		Description varchar (100) 
);

7.  create table contact_us (
		ID int primary key auto_increment , 
		Name varchar(30) , 
		Mobile_No varchar(30), 
		Email_ID varchar(30) , 
		City varchar(30) , 
		Message varchar (100) 
);
